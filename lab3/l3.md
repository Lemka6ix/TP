```python
def vstav(arr):
    for i in range(1, len(arr)):   #начинаем со 2-го, 1-ый уже отсортирован
        key = arr[i]   #текущий элемент, который мы хотим вставить
        j = i - 1   #индекс последнего элемента из отсортированной части
        while j >= 0 and arr[j] > key:  #cдвигаем элементы отсортированной части вправо, пока не найдем правильную позицию для key
            arr[j+1] = arr[j]   #сдвигаем элемент вправо
            j -= 1   #переходим к следующему элементу слева
        arr[j+1] = key   #вставляем key на правильную позицию
```

```python
def fast_sort(arr):
    """
    ИТЕРАТИВНАЯ БЫСТРАЯ СОРТИРОВКА ХОАРА
    Использует стек вместо рекурсии для избежания переполнения стека вызовов
    """
    if len(arr) < 1:
        return arr
    
    # Стек для хранения границ подмассивов, которые нужно отсортировать
    # Начинаем с всего массива: от индекса 0 до последнего элемента
    stack = [(0, len(arr) -1)]
    
    # Пока в стеке есть подмассивы для обработки
    while stack:
        # Достаем последний добавленный подмассив (LIFO - Last In First Out)
        low, high = stack.pop()

        # Если в подмассиве больше одного элемента
        if low < high:
            # Разделяем подмассив на две части относительно опорного элемента
            # cent - индекс опорного элемента после разделения
            cent = part_iter(arr, low, high)
            
            # Добавляем в стек левую часть (элементы МЕНЬШЕ опорного)
            stack.append((low, cent - 1))
            # Добавляем в стек правую часть (элементы БОЛЬШЕ опорного)
            stack.append((cent+1, high))
    
    return arr


def part_iter(arr, low, high):
    """
    ФУНКЦИЯ РАЗДЕЛЕНИЯ ХОАРА
    Переупорядочивает подмассив так, чтобы:
    - слева были элементы <= опорного
    - справа были элементы > опорного
    - возвращает конечную позицию опорного элемента
    """
    # Выбираем опорный элемент из СЕРЕДИНЫ подмассива
    # Это уменьшает вероятность худшего случая O(n²)
    mid = (low + high) // 2
    centr = arr[mid]

    # ВРЕМЕННО перемещаем опорный элемент в КОНЕЦ подмассива
    # Это нужно чтобы не мешался при обходах и обменах
    arr[mid], arr[high] = arr[high], arr[mid]
    
    # i - указатель на последний элемент в "левой" группе (<= опорного)
    # начинаем с -1, так как изначально левая группа пуста
    i = low - 1

    # j - бежит по всему подмассиву (кроме последнего элемента - там наш опорный)
    for j in range(low, high):
        # Если текущий элемент МЕНЬШЕ или РАВЕН опорному
        if arr[j] <= centr:
            # Увеличиваем границу левой группы
            i += 1
            # Обмениваем текущий элемент с первым элементом правой группы
            # Тем самым расширяем левую группу и сужаем правую
            arr[i], arr[j] = arr[j], arr[i]
    
    # ПОСЛЕДНИЙ ШАГ: возвращаем опорный элемент на правильную позицию
    # Между левой (<= опорного) и правой (> опорного) группами
    arr[i + 1], arr[high] = arr[high], arr[i + 1]
    
    # Возвращаем конечную позицию опорного элемента
    return i + 1
```
